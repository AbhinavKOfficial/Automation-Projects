class Triangle {
    int a, b, c;

    Triangle() {
        a = 3;
        b = 4;
        c = 5;
    }

    void perimeter() {
        System.out.println("Perimeter = " + (a + b + c));
    }

    void area() {
        System.out.println("Area = " + (a * b) / 2);
    }

    public static void main(String[] args) {
        Triangle t = new Triangle();
        t.perimeter();
        t.area();
    }
}
