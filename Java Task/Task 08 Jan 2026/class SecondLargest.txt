class SecondLargest {
    public static void main(String[] args) {

        int a[] = {10, 5, 20, 15};

        int largest = a[0];
        int second = a[0];

        for (int i = 0; i < 4; i++) {
            if (a[i] > largest) {
                second = largest;
                largest = a[i];
            } 
            else if (a[i] > second && a[i] != largest) {
                second = a[i];
            }
        }

        System.out.println("Second largest number = " + second);
    }
}
