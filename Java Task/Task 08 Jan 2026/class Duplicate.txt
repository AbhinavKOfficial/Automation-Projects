class Duplicate {
    public static void main(String[] args) {

        int a[] = {1, 2, 2, 3};

        for (int i = 0; i < 4; i++) {
            for (int j = i + 1; j < 4; j++) {
                if (a[i] == a[j]) {
                    System.out.println(a[i]);
                }
            }
        }
    }
}
